\usepackage[type=$if(type)$$type$$else$author-response$endif$$if(draft)$, draft$else$, final$endif$]{responseletter}

$endif$$if(authors)$  authors    = {$authors$},
$endif$$if(journal)$  journal    = {$journal$},
$endif$$if(doi)$  doi        = {$doi$},
$endif$$if(lettertitle)$  lettertitle = {$lettertitle$},
$endif$$if(cjkfont)$  cjkfont    = {$cjkfont$},
$endif$$if(cjksffont)$  cjksffont  = {$cjksffont$},
$endif$}

$for(header-includes)$
$header-includes$
$endfor$
